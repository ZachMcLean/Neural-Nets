The Parity Problem (One-step XOR - with extended time testing)

This loader is modified to provide a set of validation and testing data which is not the same length as the training data. Since we are trying to learn a solution to the parity problem and not the specific problem of sequences of a fixed length, we need to ensure we are validating/testing on data which is of a different length when calculating our metrics.

class ParityProblemDataModule(pl.LightningDataModule):
    def __init__(self,
                 batch_size = 50,
                 n_sequences = 1000,
                 n_timesteps_train = 20,
                 n_timesteps_test = 10,
                 val_split = 0.2,
                 num_workers = 4,
                 problem = 'Many2One',
                 **kwargs):
        super().__init__(**kwargs)
        self.batch_size = batch_size
        self.n_sequences = n_sequences
        self.n_timesteps = max(n_timesteps_train,n_timesteps_test)
        self.n_timesteps_train = n_timesteps_train
        self.n_timesteps_test = n_timesteps_test
        self.val_split = val_split
        self.num_workers = num_workers
        self.problem = problem
        self.data_train = None
        self.data_val = None
        self.data_test = None
        
    def setup(self, stage: str):
        if (stage == 'fit' or \
            stage == 'validate') and \
            not(self.data_train and self.data_val):
            rng = np.random.default_rng()
            X = rng.integers(0,2,size=(self.n_sequences,self.n_timesteps))
            Y = np.cumsum(X,1) % 2
            X_train = X[:,:self.n_timesteps_train]
            X_val = X[:,:self.n_timesteps_test]
            Y_train = Y[:,:self.n_timesteps_train]
            Y_val = Y[:,:self.n_timesteps_test]
            if self.problem == 'One2Many':
                X_train = X_train[:,self.n_timesteps_train-1]
                X_val = X_val[:,self.n_timesteps_test-1]
            if self.problem == 'Many2One':
                Y_train = Y_train[:,self.n_timesteps_train-1]
                Y_val = Y_val[:,self.n_timesteps_test-1]
            split_point = int(X.shape[0]*(1.0-self.val_split))
            self.data_train = list(zip(torch.Tensor(X_train[:split_point]).to(torch.long),
                                       torch.Tensor(Y_train[:split_point]).to(torch.long)))
            self.data_val = list(zip(torch.Tensor(X_val[split_point:]).to(torch.long),
                                     torch.Tensor(Y_val[split_point:]).to(torch.long)))
        if (stage == 'test' or \
            stage == 'predict') and \
            not self.data_test:
            rng = np.random.default_rng()
            X = rng.integers(0,2,size=(self.n_sequences,self.n_timesteps_test))
            Y = np.cumsum(X,1) % 2
            if self.problem == 'One2Many':
                X = X[:,self.n_timesteps_test-1]
            if self.problem == 'Many2One':
                Y = Y[:,self.n_timesteps_test-1]
            self.data_test = list(zip(torch.Tensor(X).to(torch.long),
                                      torch.Tensor(Y).to(torch.long)))
            
    def train_dataloader(self):
        return torch.utils.data.DataLoader(self.data_train,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=True)

    def val_dataloader(self):
        return torch.utils.data.DataLoader(self.data_val,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

    def test_dataloader(self):
        return torch.utils.data.DataLoader(self.data_test,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

    def predict_dataloader(self):
        return torch.utils.data.DataLoader(self.data_test,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

Two Data Loaders...

We are going to make the training sequence length (20) the same for both data loaders, but we are going to adjust the validation/testing data length and create two different data loaders. The long version may be used to validate/test on longer sequences (30), while the short version may be used to validate/test on shorter sequences (10).

Model Boilerplate

We are now ready to construct models, but we do need to provide the boilerplate classes from which to inheret all of the necessary training/validation/testing hooks, as well as metrics and optimizers, for our networks below.

# Define Trainable Module (Abstract Base Class)
class LightningBoilerplate(pl.LightningModule):
    def __init__(self, **kwargs):
        super().__init__(**kwargs) # Call the super class constructor
        
    def predict_step(self, predict_batch, batch_idx):
        x, y_true = predict_batch
        y_pred = self.predict(x)
        return y_pred, y_true

    def training_step(self, train_batch, batch_idx):
        x, y_true = train_batch
        y_pred = self(x)
        perm = (0,-1) + tuple(range(y_pred.ndim))[1:-1]
        for metric_name, metric_function in self.network_metrics.items():
            metric_value = metric_function(y_pred.permute(*perm),y_true)
            self.log('train_'+metric_name, metric_value, on_step=False, on_epoch=True)
        loss = self.network_loss(y_pred.permute(*perm),y_true)
        self.log('train_loss', loss, on_step=False, on_epoch=True)
        return loss
        
    def validation_step(self, val_batch, batch_idx):
        x, y_true = val_batch
        y_pred = self(x)
        perm = (0,-1) + tuple(range(y_pred.ndim))[1:-1]
        for metric_name, metric_function in self.network_metrics.items():
            metric_value = metric_function(y_pred.permute(*perm),y_true)
            self.log('val_'+metric_name, metric_value, on_step=False, on_epoch=True)
        loss = self.network_loss(y_pred.permute(*perm),y_true)
        self.log('val_loss', loss, on_step=False, on_epoch=True)
        return loss
        
    def test_step(self, test_batch, batch_idx):
        x, y_true = test_batch
        y_pred = self(x)
        perm = (0,-1) + tuple(range(y_pred.ndim))[1:-1]
        for metric_name, metric_function in self.network_metrics.items():
            metric_value = metric_function(y_pred.permute(*perm),y_true)
            self.log('test_'+metric_name, metric_value, on_step=False, on_epoch=True)
        loss = self.network_loss(y_pred.permute(*perm),y_true)
        self.log('test_loss', loss, on_step=False, on_epoch=True)
        return loss

# Attach loss, metrics, and optimizer
class MultiClassModule(LightningBoilerplate):
    def __init__(self,
                 num_classes,
                 **kwargs):
        super().__init__(**kwargs) # Call the super class constructor
        self.num_classes = num_classes

        # This creates an accuracy function
        self.network_metrics = torch.nn.ModuleDict({
            'acc': torchmetrics.classification.Accuracy(task='multiclass',
                                                        num_classes=num_classes)
        })
        # This creates a loss function
        self.network_loss = torch.nn.CrossEntropyLoss()

    def configure_optimizers(self):
        optimizer = torch.optim.Adam(self.parameters(), lr=0.001)
        return optimizer

class Many2OneNetwork(MultiClassModule):
    def __init__(self,
                 num_input_tokens,
                 hidden_size = 64,
                 **kwargs):
        super().__init__(**kwargs)
        self.save_hyperparameters()
        
        self.embedding = torch.nn.Embedding(num_input_tokens,
                                            hidden_size)
        self.rnn_layer = torch.nn.RNN(hidden_size,
                                      hidden_size)
        self.output_layer = torch.nn.Linear(hidden_size,
                                            self.num_classes)

    def forward(self, x):
        y = x
        y = self.embedding(y)
        y = self.rnn_layer(y.permute(1,0,2)) # Returns 2 tensors (as-tuple)
        y = self.output_layer(y[0][-1,:]) # Keep only the last
                                          # time-step...
        return y
