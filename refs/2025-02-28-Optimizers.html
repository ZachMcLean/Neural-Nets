class MulticlassClassificationDataModule(pl.LightningDataModule):
    def __init__(self,
                 data_url = "https://jupyterhub.cs.mtsu.edu/azuread/services/csci4850-materials/public/WDBC.txt",
                 batch_size = 32,
                 val_split = 0.2,
                 num_workers = 4,
                 **kwargs):
        super().__init__(**kwargs)
        self.data_url = data_url
        self.batch_size = batch_size
        self.val_split = val_split
        self.num_workers = num_workers
        self.data_train = None
        self.data_val = None

    def setup(self, stage: str):
        if (stage == 'fit' or \
            stage == 'validate' or \
            stage == 'predict') and \
            not(self.data_train and self.data_val):
            rng = np.random.default_rng()
            data = np.loadtxt(self.data_url)
            rng.shuffle(data)
            split_point = int(data.shape[0]*(1.0-self.val_split))
            self.data_train = list(zip(torch.tensor(data[:split_point,:-1]).to(torch.float32),
                                       torch.tensor(data[:split_point,-1]).to(torch.long)))
            self.data_val = list(zip(torch.tensor(data[split_point:,:-1]).to(torch.float32),
                                     torch.tensor(data[split_point:,-1]).to(torch.long)))

    def train_dataloader(self):
        return torch.utils.data.DataLoader(self.data_train,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=True)

    def val_dataloader(self):
        return torch.utils.data.DataLoader(self.data_val,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

    def predict_dataloader(self):
        return torch.utils.data.DataLoader(self.data_val,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

class ResidualLayer(pl.LightningModule):
    def __init__(self,
                 size,
                 **kwargs):
        super().__init__(**kwargs)
        self.net_input = torch.nn.Linear(size,
                                         size)
        self.activation = torch.nn.Tanh()
    def forward(self, x):
        y = x
        y = self.net_input(y)
        y = self.activation(y)
        y = x + y
        return y

class NeuralNetwork(pl.LightningModule):
    def __init__(self,**kwargs):
        super().__init__(**kwargs)
        # Linear Projection
        self.projection = torch.nn.Linear(30,100)
        # Stacked Residual Layers
        self.hidden = torch.nn.Sequential(*[ResidualLayer(100) for _ in range(5)])
        # Output Logits
        self.output = torch.nn.Linear(100,2)
        # Activation Function (for predict() below)
        self.output_activation = torch.nn.Softmax(dim=-1)
        self.mca = torchmetrics.classification.Accuracy(task='multiclass',
                                                        num_classes=2)
        self.cce = torch.nn.CrossEntropyLoss()
        
    def forward(self, x):
        y = x
        # Linear Projection
        y = self.projection(y)
        # Stacked Residual Layers
        y = self.hidden(y)
        # Output Logits
        y = self.output(y)
        return y

    def predict(self, x):
        y = self(x)
        y = self.output_activation(y)
        return y

    def configure_optimizers(self):
        # Here's where the magic happens!
        # Can only pick one-at-a-time.
        
        # optimizer = torch.optim.SGD(self.parameters(), lr=1e-3)
        # optimizer = torch.optim.RMSprop(self.parameters(), lr=1e-3)
        # optimizer = torch.optim.Adagrad(self.parameters(), lr=1e-3)
        # optimizer = torch.optim.Adam(self.parameters(), lr=1e-3)
        # optimizer = torch.optim.AdamW(self.parameters(), lr=1e-3)
        return optimizer

    def predict_step(self, predict_batch, batch_idx):
        x, y_true = predict_batch
        y_pred = self.predict(x)
        return y_pred, y_true

    def training_step(self, train_batch, batch_idx):
        x, y_true = train_batch
        y_pred = self(x)
        loss = self.cce(y_pred,y_true)
        acc = self.mca(y_pred,y_true)
        self.log('train_loss', loss, on_step=False, on_epoch=True)
        self.log('train_acc', acc, on_step=False, on_epoch=True)
        return loss

    def validation_step(self, val_batch, batch_idx):
        x, y_true = val_batch
        y_pred = self(x)
        loss = self.cce(y_pred,y_true)
        acc = self.mca(y_pred,y_true)
        self.log('val_loss', loss, on_step=False, on_epoch=True)
        self.log('val_acc', acc, on_step=False, on_epoch=True)
        return loss

    def test_step(self, test_batch, batch_idx):
        x, y_true = test_batch
        y_pred = self(x)
        loss = self.cce(y_pred,y_true)
        acc = self.mca(y_pred,y_true)
        self.log('test_loss', loss, on_step=False, on_epoch=True)
        self.log('test_acc', acc, on_step=False, on_epoch=True)
        return loss
