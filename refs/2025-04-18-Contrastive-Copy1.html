class ContrastiveDataset(torch.utils.data.Dataset):
    def __init__(self, x, y, **kwargs):
        super().__init__(**kwargs)
        self.x = torch.Tensor(x)
        self.y = torch.Tensor(y).long()
        # Group by labels
        groups = [[] for _ in range(len(torch.unique(self.y)))]
        for x_i, y_i in zip(self.x, self.y):
            groups[y_i].append(x_i)
        # print(groups)
        self.groups = groups
        # self.groups = list(map(torch.Tensor, groups))
        
    def __len__(self):
        return len(self.x)

    def __getitem__(self, idx):
        return self.x[idx], self.groups[self.y[idx]][torch.randint(low=0,high=len(self.groups[self.y[idx]]),size=(1,))]

class DenseLayer(pl.LightningModule):
    def __init__(self,
                 in_features,
                 out_features,
                 activation=None, **kwargs):
        super().__init__(**kwargs)
        self.layer = torch.nn.Sequential(
            torch.nn.Linear(in_features,out_features),
            activation
        ) if activation is not None else torch.nn.Sequential(
            torch.nn.Linear(in_features,out_features),
        )

    def forward(self, x):
        y = self.layer(x)
        return y

class ContrastiveMultiLayerNetwork(pl.LightningModule):
    def __init__(self,
                 input_size,
                 embedding_size,
                 max_batch_size,
                 **kwargs):
        super().__init__(**kwargs)
        self.save_hyperparameters()
        self.hidden_layer = DenseLayer(input_size,embedding_size,activation=torch.nn.SiLU())
        self.output_layer = DenseLayer(embedding_size,embedding_size)
        self.mca = multiclass_accuracy
        self.cce = torch.nn.CrossEntropyLoss()

    def forward(self, y1, y2):
        y1 = self.output_layer(self.hidden_layer(y1))
        y2 = self.output_layer(self.hidden_layer(y2))
        y = y1 @ y2.t()
        return y, y1, y2

    def embed(self, x):
        return self.output_layer(self.hidden_layer(x))

    def configure_optimizers(self):
        optimizer = torch.optim.Adam(self.parameters(), lr=1e-3)
        return optimizer

    def training_step(self, train_batch, batch_idx):
        y_pred, _, _ = self(train_batch[0],train_batch[1])
        y_true = torch.arange(len(y_pred),requires_grad=False).type_as(y_pred).long()
        loss1 = self.cce(y_pred, y_true)
        loss2 = self.cce(y_pred.t(), y_true)
        loss = (loss1+loss2)/2.0
        self.log('train_loss', loss, on_step=False, on_epoch=True)
        acc1 = self.mca(y_pred, y_true, num_classes=y_pred.shape[0])
        acc2 = self.mca(y_pred.t(), y_true, num_classes=y_pred.shape[0])
        acc = (acc1+acc2)/2.0
        self.log('train_acc', acc, on_step=False, on_epoch=True)
        return loss

    def validation_step(self, val_batch, batch_idx):
        y_pred, y1_pred, _ = self(val_batch[0], val_batch[1])
        y_true = torch.arange(len(y_pred),requires_grad=False).type_as(y_pred).long()
        loss1 = self.cce(y_pred, y_true)
        loss2 = self.cce(y_pred.t(), y_true)
        loss = (loss1+loss2)/2.0
        self.log('val_loss', loss, on_step=False, on_epoch=True)
        acc1 = self.mca(y_pred, y_true, num_classes=y_pred.shape[0])
        acc2 = self.mca(y_pred.t(), y_true, num_classes=y_pred.shape[0])
        acc = (acc1+acc2)/2.0
        self.log('val_acc', acc, on_step=False, on_epoch=True)
        return loss

The value above represents the result of using a nearest-neighbor interpolation. All validation data points are compared to all training data points. The class label of the training data point with the closest embedding to each validation data point embedding is used to provide a predicted label for each validation data point. These are then compared to the actual targets/labels to obtain the final accuracy assessment.

Above, we are plotting the embeddings for both basis vectors and validation vectors onto the same 2D projection using PCA. Circles indicate the basis vector locations (a reasonable number of human-labeled points) and Xs indicate the validation vector locations. If we assume a nearest-neighbor approach is reasonable for our data, each validation vector is predicted to be the same class as the closest basis vector. For a few validation vectors, this approach leads to misclassification. However, this exercise illustrates the concept of how just having a few labeled data vectors can then be used to predict good classification labels for unlabeled data in the future.
