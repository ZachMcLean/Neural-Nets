# Define model
class NeuralNetwork(pl.LightningModule):
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.net_input = torch.nn.Linear(30,1)
        self.activation = torch.nn.Sigmoid()
        self.b_acc = torchmetrics.classification.Accuracy(task='binary')
        self.bce_loss = torch.nn.BCELoss()
        
    def forward(self, x):
        y = x
        y = self.net_input(y)
        y = self.activation(y)
        return y

    def configure_optimizers(self):
        optimizer = torch.optim.SGD(self.parameters(), lr=10.0)
        return optimizer

    def predict_step(self, pred_batch, batch_idx):
        x, y_true = pred_batch
        y_pred = self(x)
        return y_pred, y_true
        
    def training_step(self, train_batch, batch_idx):
        x, y_true = train_batch
        y_pred = self(x)
        acc = self.b_acc(y_pred,y_true)
        loss = self.bce_loss(y_pred,y_true)
        self.log('train_acc', acc, on_step=False, on_epoch=True)
        self.log('train_loss', loss, on_step=False, on_epoch=True)
        return loss
        
model = NeuralNetwork()
print(model)

import sympy as sp

# Use sympy for differentiation of our activation
# functions and error function
x = sp.symbols('x')
y = sp.symbols('y')

# Tanh Activation Function - Vectorized g(x)
g = (sp.exp(2*x)-1)/(sp.exp(2*x)+1)
tanh = np.vectorize(lambda x: np.float32(g.evalf(subs=dict(x=x))))

# First Derivative of Tanh - Vectorized dg(x)/dx
dgdx = g.diff(x)
dtanh = np.vectorize(lambda x: np.float32(dgdx.evalf(subs=dict(x=x))))

# Sigmoid Activation Function - Vectorized f(x)
f = 1 / (sp.exp(-x)+1)
sigmoid = np.vectorize(lambda x: np.float32(f.evalf(subs=dict(x=x))))

# First Derivative of Tanh - Vectorized dg(x)/dx
dfdx = f.diff(x)
dsigmoid = np.vectorize(lambda x: np.float32(dfdx.evalf(subs=dict(x=x))))

# Binary Cross-entropy Error (and first derivative)
b = -((y*sp.log(x)+(1-y)*sp.log(1-x)))
dbdx = b.diff(x)

# Vectorized forms of BCE and dBCE(x)/x
bce = np.vectorize(lambda x,y: np.float32(b.evalf(subs=dict(x=x,y=y))))
dbce = np.vectorize(lambda x,y: np.float32(dbdx.evalf(subs=dict(x=x,y=y))))

Complete Training Regime for WDBC (for Binary Classification)

The code below implements:

  1. a single-layer neural network with a single sigmoid unit output layer
  2. two data loaders for creating a training data set (80%) and validation data set (20%)
  3. a trainer and logger for executing the training/validation (and prediction) process
  4. the use of matplotlib to plot the resulting training/validation loss and accuracy curves

# (Re)define model - duplicated from above
class NeuralNetwork(pl.LightningModule):
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.net_input = torch.nn.Linear(30,1)
        self.activation = torch.nn.Sigmoid()
        self.b_acc = torchmetrics.classification.Accuracy(task='binary')
        self.bce_loss = torch.nn.BCELoss()
        
    def forward(self, x):
        y = x
        y = self.net_input(y)
        y = self.activation(y)
        return y

    def configure_optimizers(self):
        optimizer = torch.optim.SGD(self.parameters(), lr=0.1)
        return optimizer

    def predict_step(self, pred_batch, batch_idx):
        x, y_true = pred_batch
        y_pred = self(x)
        return y_pred, y_true
        
    def training_step(self, train_batch, batch_idx):
        x, y_true = train_batch
        y_pred = self(x)
        acc = self.b_acc(y_pred,y_true)
        loss = self.bce_loss(y_pred,y_true)
        self.log('train_acc', acc, on_step=False, on_epoch=True)
        self.log('train_loss', loss, on_step=False, on_epoch=True)
        return loss
        
    def validation_step(self, val_batch, batch_idx):
        x, y_true = val_batch
        y_pred = self(x)
        acc = self.b_acc(y_pred,y_true)
        loss = self.bce_loss(y_pred,y_true)
        self.log('val_acc', acc, on_step=False, on_epoch=True)
        self.log('val_loss', loss, on_step=False, on_epoch=True)
        return loss

Data Loader Preparation

Data loaders wrap a data set to provide mechanisms for easily iterating over batches of feature vectors (and associated targets), and reshuffling for each epoch. While we could write custom code to iterate over the data ourselves, we instead utilize DataLoader since it will save use from having to write this custom (really more boilerplate) code. This prevents bugs, saves time, and keeps our own code more concise and readable (once we understand what functionality it provides).

A typical rule-of-thumb for setting up training and validation data is to:

  1. shuffle all rows in the data set
  2. use the top rows (80%) of the data for training
  3. use the bottom rows (20%) of the data for validation

This 80%/20% rule is just something often works well in practice. Information theoretic methods for calculating "optimal" splits for generalization purposes have been introduced in the field, but are not widely adopted because (most of the time) the calculated values are often close to 80%/20%. When they are not, it is typically because the data set has particular properties where we would already be able to guess that the 80%/20% rule is not appropriate (for example, a small data set where one class is not very common - might lead to none of these uncommon examples being in the validation data resulting in biased performance).

How do we use a DataLoader?

Data loaders provide iterators over a data set. We might have split our data set into training/validation, but during the training process we muct shuffle our training data and break it into groups of batch_size (32 above) feature vector/target pairs. The data loader let's us do that.

class BinaryClassificationDataModule(pl.LightningDataModule):
    def __init__(self,
                 data_url = "https://jupyterhub.cs.mtsu.edu/azuread/services/csci4850-materials/public/WDBC.txt",
                 batch_size = 32,
                 val_split = 0.2,
                 num_workers = 4,
                 **kwargs):
        super().__init__(**kwargs)
        self.data_url = data_url
        self.batch_size = batch_size
        self.val_split = val_split
        self.num_workers = num_workers
        self.data_train = None
        self.data_val = None

    def setup(self, stage: str):
        if (stage == 'fit' or \
            stage == 'validate' or \
            stage == 'predict') and \
            not(self.data_train and self.data_val):
            rng = np.random.default_rng()
            data = np.loadtxt(self.data_url)
            rng.shuffle(data)
            split_point = int(data.shape[0]*(1.0-self.val_split))
            self.data_train = list(zip(torch.tensor(data[:split_point,:-1]).to(torch.float32),
                                       torch.tensor(data[:split_point,-1:]).to(torch.float32)))
            self.data_val = list(zip(torch.tensor(data[split_point:,:-1]).to(torch.float32),
                                     torch.tensor(data[split_point:,-1:]).to(torch.float32)))

    def train_dataloader(self):
        return torch.utils.data.DataLoader(self.data_train,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=True)

    def val_dataloader(self):
        return torch.utils.data.DataLoader(self.data_val,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

    def predict_dataloader(self):
        return torch.utils.data.DataLoader(self.data_val,
                                           batch_size=self.batch_size,
                                           num_workers=self.num_workers,
                                           shuffle=False)

Loggers and Trainers

The Trainer will provide the functionality needed to iterate over the training batches, calculate the loss values and derivatives, and then update the weights accordingly. It also provides a way to iterate over the validation data to check for generalization while the training process is underway. The documentation shows the general scaffold for how trainers operate, but keep in mind that it does do a few other things as well: https://lightning.ai/docs/pytorch/stable/common/trainer.html#under-the-hood

The Logger provides the trainer with an interface for collecting performance measure statistics that we are calculating during the training process. We will be utilizing comma separated value (CSV) files (a text-only spreadsheet format), but there are other loggers which can write to web-based platforms in the cloud for managing and accessing this data. For now, we will access our CSV files after the training process is complete, and create learning curves from the collected data. While we won't be using other loggers in the course, they can be easily integrated by choosing the appropriate logger class. Here is more information on the possible logger options: https://lightning.ai/docs/pytorch/stable/api_references.html#loggers
